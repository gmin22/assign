import scala.util.Random

// Function to convert a decimal number to binary
def decimalToBinary(n: Int): String = {
  var binary = ""
  var num = n
  while (num > 0) {
    binary = (num % 2).toString + binary
    num = num / 2
  }
  binary
}

// Function to convert a decimal number to octal
def decimalToOctal(n: Int): String = {
  var octal = ""
  var num = n
  while (num > 0) {
    octal = (num % 8).toString + octal
    num = num / 8
  }
  octal
}

// Read in five random numbers
val randomNumbers = for (i <- 1 to 5) yield Random.nextInt(100)

// Convert each number to binary and octal and print the results
for (num <- randomNumbers) {
  val binary = decimalToBinary(num)
  val octal = decimalToOctal(num)
  println(s"$num in binary: $binary")
  println(s"$num in octal: $octal")
}

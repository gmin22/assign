//SLIP2
object StringManipulation {
  def main(args: Array[String]): Unit = {
    // Read in the two strings
    print("Enter the first string: ")
    val firstString = scala.io.StdIn.readLine()
    print("Enter the second string: ")
    val secondString = scala.io.StdIn.readLine()

    // Find the occurrence of the second string in the first string
    val index = firstString.indexOf(secondString)

    // Reverse the case of the second string in the first string
    val result = if (index == -1) {
      // Second string not found, return the original first string
      firstString
    } else {
      // Second string found, reverse the case of each character
      val reversed = secondString.map(c => if (c.isUpper) c.toLower else c.toUpper)
      // Replace the occurrence of the second string in the first string with the reversed string
      firstString.replace(secondString, reversed)
    }

    // Print the resulting string
    println(result)
  }
}

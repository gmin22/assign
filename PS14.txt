// Create the first set of strings
val set1 = Set("hello", "world", "goodbye", "universe")

// Create the second set of strings
val set2 = Set("hello", "universe", "wonderland", "hello")

// Find the common strings between the two sets
val common = set1.intersect(set2)

// Remove the common strings from the two sets
val set1WithoutCommon = set1 -- common
val set2WithoutCommon = set2 -- common

// Merge the two sets
val mergedSet = set1WithoutCommon ++ set2WithoutCommon

// Display the resultant set
println(mergedSet)

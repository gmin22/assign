def reverseCase(s: String): String = {
  // Define a function that takes a character and returns the character
  // with its case reversed
  def reverseCaseChar(c: Char): Char = {
    if (c.isUpper) c.toLower
    else if (c.isLower) c.toUpper
    else c
  }

  // Use the map function to apply the reverseCaseChar function to each character in the string
  s.map(reverseCaseChar)
}

// Call the reverseCase function using the MAP function
val reversedCases = List("Hello", "WORLD", "GoOdByE").map(reverseCase)

// Display the result
println(reversedCases)
